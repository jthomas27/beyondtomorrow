<script>
(function () {
  var celebrated = false;

  function showModal() {
    var overlay = document.getElementById('bt-welcome-overlay');
    if (!overlay) return;
    overlay.classList.add('bt-visible');
    var close = document.getElementById('bt-welcome-close');
    var dismiss = document.getElementById('bt-welcome-dismiss');
    function hide() { overlay.classList.remove('bt-visible'); }
    close.addEventListener('click', hide);
    dismiss.addEventListener('click', hide);
    overlay.addEventListener('click', function (e) { if (e.target === overlay) hide(); });
    document.addEventListener('keydown', function onKey(e) {
      if (e.key === 'Escape') { hide(); document.removeEventListener('keydown', onKey); }
    });
  }

  function celebrate() {
    if (celebrated) return;
    celebrated = true;
    confetti({ particleCount: 100, angle: 60, spread: 65, origin: { x: 0, y: 0.6 }, colors: ['#c8b8ff', '#a78bfa', '#818cf8', '#6366f1', '#ffd6a5', '#f9a8d4'] });
    confetti({ particleCount: 100, angle: 120, spread: 65, origin: { x: 1, y: 0.6 }, colors: ['#c8b8ff', '#a78bfa', '#818cf8', '#6366f1', '#ffd6a5', '#f9a8d4'] });
    setTimeout(function () {
      confetti({ particleCount: 60, spread: 80, origin: { x: 0.5, y: 0.4 }, colors: ['#c8b8ff', '#a78bfa', '#ffd6a5'] });
    }, 400);
    showModal();
    setTimeout(function () { celebrated = false; }, 15000);
  }

  /* Detect signup success by watching the Portal iframe DOM.
     Handles two distinct Ghost Portal UI states:
       1. New subscriber success PAGE — after magic-link email is sent,
          Portal replaces the form with ".gh-portal-main-title" containing
          "Please check your inbox." (Ghost 5.x Portal 2.x default text).
       2. Logged-in member toast NOTIFICATION — ".gh-portal-notification-icon.success"
          shown after profile updates etc.
     This approach is read-only — does NOT override window.fetch or any global. */
  function scanIframeForSuccess(iframe) {
    try {
      var doc = iframe.contentDocument;
      if (!doc || !doc.body) return false;

      // 1. New subscriber success page — main title says "check your inbox"
      var title = doc.querySelector('.gh-portal-main-title');
      if (title) {
        var t = title.textContent.toLowerCase();
        if (t.indexOf('inbox') !== -1 || t.indexOf('check your email') !== -1 || t.indexOf('sent you') !== -1) return true;
      }

      // 2. Broader heading scan in case Ghost changes the title class
      // h1 included — Ghost Portal main-title is typically an h1
      var headings = doc.querySelectorAll('h1, h2, h3');
      for (var h = 0; h < headings.length; h++) {
        var ht = headings[h].textContent.toLowerCase();
        if (ht.indexOf('inbox') !== -1 || ht.indexOf('check your email') !== -1) return true;
      }

      // 3. Logged-in member toast notification (original selectors — kept as fallback)
      var el = doc.querySelector('.gh-portal-notification-icon.success');
      if (el) return true;
      var ps = doc.querySelectorAll('.gh-portal-notification p');
      for (var i = 0; i < ps.length; i++) {
        if (ps[i].textContent.indexOf('Check your email') !== -1) return true;
      }
    } catch (e) { /* cross-origin or not ready */ }
    return false;
  }

  function checkPortal() {
    var root = document.getElementById('ghost-portal-root');
    if (!root) return;
    var iframes = root.querySelectorAll('iframe');
    for (var i = 0; i < iframes.length; i++) {
      if (scanIframeForSuccess(iframes[i])) {
        celebrate();
        return;
      }
    }
  }

  /* Observe DOM mutations on the portal root and its iframe contents.
     This fires whenever Portal updates its UI (e.g. showing the success
     notification after a successful magic-link request). */
  var iframeObservers = [];

  function watchIframe(iframe) {
    if (iframe._btWatched) return;
    iframe._btWatched = true;
    function attach() {
      try {
        var doc = iframe.contentDocument;
        if (!doc) return;
        var target = doc.body || doc.documentElement;
        if (!target) return;
        var obs = new MutationObserver(checkPortal);
        obs.observe(target, { childList: true, subtree: true });
        iframeObservers.push(obs);
      } catch (e) { /* cross-origin */ }
    }
    attach();
    iframe.addEventListener('load', attach);
  }

  function watchPortalRoot() {
    var root = document.getElementById('ghost-portal-root');
    if (!root || root._btRootWatched) return;
    root._btRootWatched = true;

    new MutationObserver(function () {
      var iframes = root.querySelectorAll('iframe');
      for (var j = 0; j < iframes.length; j++) watchIframe(iframes[j]);
      checkPortal();
    }).observe(root, { childList: true, subtree: true });
  }

  /* Watch document.body for the portal root div to appear. */
  function init() {
    watchPortalRoot();
    new MutationObserver(watchPortalRoot)
      .observe(document.body, { childList: true, subtree: false });
  }

  if (document.body) init();
  else document.addEventListener('DOMContentLoaded', init);
})();
</script>
